import * as params from '@params';

function genTab() {
  console.log(params.title);
  var postTab = document.createElement("div");
  postTab.id = params.title;
  postTab.style.borderLeft = "3px solid #a89984";
  postTab.style.paddingLeft = "3px";
  postTab.id = params.relPermalink;
  if (!document.getElementById(params.relPermalink)) {
    document.getElementById("content").appendChild(postTab);
    updateTab(params.relPermalink);
  }

}

function updateTab(pageLink) {
  document.getElementById(pageLink).innerHTML = params.content;
}
